**Goal:** Replace hand-aligned `<pre>` chord/lyric song pages with a ChordPro source format + a Python renderer in `py-gen.py`, auto-migrate all 52 songs, and add chord transpose + user font-size control.

**Architecture:** Each song dir holds `song.pro` (ChordPro source). `py-gen.py` renders each `song.pro` to `index.html`, then regenerates the root search list + `sw.js` cache list and bumps the cache version. A one-off `convert_to_chordpro.py` migrates existing pages; a verification script proves the conversion preserved every chord/lyric token. Transpose + font-size are client-side JS in the rendered template.

**Tech Stack:** Python 3 (stdlib only — no pytest, tests are plain `assert` scripts run with `python3`), vanilla JS, CSS. Playwright (MCP) for in-browser JS verification. Deploy via existing `deploy.sh` (rsync to nginx).

# chordpro.py
"""ChordPro parsing + HTML rendering for the songbook. Stdlib only."""
import re
import html as _html

_DIRECTIVE = re.compile(r'^\{(\w+):\s*(.*?)\s*\}$')
_CHORD = re.compile(r'\[([^\]]*)\]')


def parse_pro(text):
    """ChordPro text -> {'title','artist','lines':[...]}.

    Line types:
      {'type':'blank'}
      {'type':'chordonly','chords':[...]}
      {'type':'lyric','chunks':[{'chord','text'},...]}
    """
    title = artist = ""
    lines = []
    for raw in text.split("\n"):
        line = raw.rstrip("\r")
        m = _DIRECTIVE.match(line.strip())
        if m:
            key, val = m.group(1).lower(), m.group(2)
            if key == "title":
                title = val
            elif key == "artist":
                artist = val
            continue
        if line.strip() == "":
            lines.append({"type": "blank"})
            continue
        # chord-only line: stripping all [..] tokens leaves only whitespace/punctuation
        stripped = _CHORD.sub("", line)
        if stripped.strip(" ,|:") == "" and _CHORD.search(line):
            chords = [c.strip() for c in _CHORD.findall(line) if c.strip()]
            # a token may itself contain comma-separated chords e.g. "D, A7"
            flat = []
            for c in chords:
                flat.extend(p.strip() for p in c.split(",") if p.strip())
            lines.append({"type": "chordonly", "chords": flat})
            continue
        lines.append({"type": "lyric", "chunks": _split_chunks(line)})
    return {"title": title, "artist": artist, "lines": lines}


def _split_chunks(line):
    chunks = []
    pos = 0
    leading = True
    for m in _CHORD.finditer(line):
        text_before = line[pos:m.start()]
        if leading:
            if text_before != "":
                chunks.append({"chord": "", "text": text_before})
            leading = False
        else:
            chunks[-1]["text"] = text_before
        chunks.append({"chord": m.group(1), "text": ""})
        pos = m.end()
    tail = line[pos:]
    if not chunks:
        return [{"chord": "", "text": line}]
    chunks[-1]["text"] = tail
    return chunks
```

def _esc(s):
    return _html.escape(s, quote=True)


def render_song_body(lines):
    out = []
    for ln in lines:
        if ln["type"] == "blank":
            out.append("")
            continue
        if ln["type"] == "chordonly":
            spans = "".join(
                f'<span class="c" data-chord="{_esc(c)}">{_esc(c)}</span>' for c in ln["chords"]
            )
            out.append(f'<div class="chordrow">{spans}</div>')
            continue
        # lyric
        spans = []
        for ch in ln["chunks"]:
            spans.append(
                f'<span class="ch"><span class="c" data-chord="{_esc(ch["chord"])}">'
                f'{_esc(ch["chord"])}</span>{_esc(ch["text"])}</span>'
            )
        out.append(f'<div class="line">{"".join(spans)}</div>')
    return "\n".join(out)
```

# convert_to_chordpro.py
"""One-off: convert space-aligned <pre> song pages to song.pro (ChordPro). Stdlib only."""
import glob
import os
import re
import html as _html

_TITLE = re.compile(r'<title>(.*?)</title>', re.S | re.I)
_ARTIST = re.compile(r'<p><b><pre class="yellow"[^>]*>(.*?)</pre>', re.S)
_INNER_CHORD = re.compile(r'<pre class="yellow"[^>]*>(.*?)</pre>', re.S)


def extract_title(src):
    m = _TITLE.search(src)
    return _html.unescape(m.group(1).strip()) if m else ""


def extract_artist(src):
    m = _ARTIST.search(src)
    return _html.unescape(re.sub(r"<[^>]+>", "", m.group(1)).strip()) if m else ""


def merge(chord_line, lyric_line):
    """Insert [chord] tokens into lyric at the chord's column. Returns ChordPro line."""
    tokens = []  # (col, chord)
    for m in re.finditer(r'\S+', chord_line):
        tokens.append((m.start(), m.group(0)))
    result = list(lyric_line)
    # insert from rightmost so earlier indices stay valid
    for col, chord in sorted(tokens, reverse=True):
        ins = f"[{chord}]"
        if col >= len(result):
            result = result + list(" " * (col - len(result)))  # pad if past end
            result.append(ins)
        else:
            result.insert(col, ins)
    return "".join(result).rstrip()


def _song_body(src):
    """Return the inner content of the main song <div><pre>...</pre></div> as raw text
    with inner chord <pre> markers preserved as lines starting with \x00."""
    # main body is the second top-level <div>; grab from after author block
    after = src.split("</div>", 1)[1] if "</div>" in src else src
    m = re.search(r"<pre>(.*)</pre>\s*</div>", after, re.S)
    body = m.group(1) if m else ""
    # replace inner chord pres with a sentinel line: \x00 + chord text
    body = _INNER_CHORD.sub(lambda mm: "\x00" + _html.unescape(mm.group(1)), body)
    return _html.unescape(body)


def convert(src):
    title = extract_title(src)
    artist = extract_artist(src)
    body = _song_body(src)
    raw_lines = body.split("\n")
    out = []
    i = 0
    while i < len(raw_lines):
        ln = raw_lines[i]
        if ln.startswith("\x00"):
            chord_line = ln[1:]
            nxt = raw_lines[i + 1] if i + 1 < len(raw_lines) else ""
            if nxt and not nxt.startswith("\x00") and nxt.strip() != "":
                out.append(merge(chord_line, nxt))
                i += 2
                continue
            # chord-only line
            chords = re.findall(r'\S+', chord_line)
            out.append(" ".join(f"[{c}]" for c in chords))
            i += 1
            continue
        out.append(ln.rstrip())
        i += 1
    # collapse 3+ blank lines to max 1, trim edges
    pro = []
    for ln in out:
        if ln.strip() == "" and (not pro or pro[-1] == ""):
            continue
        pro.append(ln)
    while pro and pro[-1] == "":
        pro.pop()
    header = f"{{title: {title}}}\n{{artist: {artist}}}\n\n"
    return header + "\n".join(pro) + "\n"


def main():
    for page in sorted(glob.glob("[0-9][0-9][0-9][0-9]/*/index.html")):
        src = open(page, encoding="utf-8").read()
        pro = convert(src)
        out = os.path.join(os.path.dirname(page), "song.pro")
        open(out, "w", encoding="utf-8").write(pro)
        print("wrote", out)


if __name__ == "__main__":
    main()
```

# verify_conversion.py
"""Assert each rendered song.pro preserves the original page's chord+lyric tokens."""
import glob, os, re, html as _html
import chordpro

def tokens_from_original(src):
    after = src.split("</div>", 1)[1]
    m = re.search(r"<pre>(.*)</pre>\s*</div>", after, re.S)
    body = _html.unescape(re.sub(r'<pre class="yellow"[^>]*>(.*?)</pre>',
                                 lambda mm: " " + mm.group(1) + " ", m.group(1), flags=re.S))
    return re.findall(r'\S+', body)

def tokens_from_pro(pro_text):
    doc = chordpro.parse_pro(pro_text)
    parts = []
    for ln in doc["lines"]:
        if ln["type"] == "chordonly":
            parts.extend(ln["chords"])
        elif ln["type"] == "lyric":
            for ch in ln["chunks"]:
                if ch["chord"]:
                    parts.append(ch["chord"])
                parts.extend(ch["text"].split())
    return parts

def main():
    bad = []
    for page in sorted(glob.glob("[0-9][0-9][0-9][0-9]/*/index.html")):
        d = os.path.dirname(page)
        pro = os.path.join(d, "song.pro")
        if not os.path.exists(pro):
            bad.append((d, "no song.pro")); continue
        a = tokens_from_original(open(page, encoding="utf-8").read())
        b = tokens_from_pro(open(pro, encoding="utf-8").read())
        if a != b:
            # find first divergence
            i = next((k for k in range(min(len(a), len(b))) if a[k] != b[k]), min(len(a), len(b)))
            bad.append((d, f"diverge at {i}: orig={a[i:i+4]} pro={b[i:i+4]}"))
    if bad:
        print("MISMATCHES:", len(bad))
        for d, why in bad:
            print(" ", d, "->", why)
    else:
        print("ALL 52 MATCH")
    return 1 if bad else 0

if __name__ == "__main__":
    import sys; sys.exit(main())
```

<script>
(function(){
  var SHARP=["C","C#","D","D#","E","F","F#","G","G#","A","B","H"]; // Czech: B=Bb, H=B-natural
  var IDX={"C":0,"C#":1,"DB":1,"D":2,"D#":3,"EB":3,"E":4,"FB":4,"F":5,"F#":6,"GB":6,
           "G":7,"G#":8,"AB":8,"A":9,"A#":10,"B":10,"HB":10,"H":11,"CB":11};
  function shift(token, n){
    // token like D, Hm, A7, C#maj7, D/F#
    var m=token.match(/^([A-H][b#]?)(.*)$/); if(!m) return token;
    var root=m[1], rest=m[2];
    var key=root.toUpperCase();
    if(!(key in IDX)) return token;
    var base=SHARP[((IDX[key]+n)%12+12)%12];
    // handle slash bass in rest
    rest=rest.replace(/\/([A-H][b#]?)/, function(_,b){
      var bk=b.toUpperCase(); if(!(bk in IDX)) return "/"+b;
      return "/"+SHARP[((IDX[bk]+n)%12+12)%12];
    });
    return base+rest;
  }
  window.__shift=shift; // exposed for tests
  var off=0;
  var key="trans:"+location.pathname;
  try{ off=parseInt(localStorage.getItem(key)||"0",10)||0; }catch(e){}
  function apply(){
    document.querySelectorAll(".c[data-chord]").forEach(function(el){
      var orig=el.getAttribute("data-chord");
      el.textContent = orig ? shift(orig, off) : "";
    });
    try{ localStorage.setItem(key,String(off)); }catch(e){}
  }
  function bump(d){ off=((off+d)%12+12)%12; apply(); }
  document.getElementById("transUp").addEventListener("click",function(){bump(1);});
  document.getElementById("transDown").addEventListener("click",function(){bump(-1);});

})();
</script>
